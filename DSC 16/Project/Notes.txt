Second code -

    K = 100;

    payoff = max(K - S,0);

    figure
    plot(S,V,'b','LineWidth',2)
    hold on
    plot(S,payoff,'r--','LineWidth',2)

    legend('Option Value','Payoff')
    xlabel('Stock Price')
    ylabel('Value')
    title('American Put vs Payoff')
    grid on
